git clone --recursive https://github.com/wxhere/docker-ubuntu-vnc-desktop.git ./docker-ubuntu-vnc-desktop
git submodule update --init --recursive

sudo docker build -f Dockerfile.amd64 -t wxhere/ubuntu-desktop-lxde-vnc:lxqt-amd64 .

docker run -d --rm \
  --name=ubuntu-desktop-lxqt-test \
  -e VNC_PASSWORD=123456 \
  -e HTTP_PASSWORD=123456 \
  -e USER=sandbox \
  -e PASSWORD=123456 \
  -p 5900:5900 \
  -p 6080:80 \
  wxhere/ubuntu-desktop-lxde-vnc:lxqt-amd64

docker push wxhere/ubuntu-desktop-lxde-vnc:lxqt-amd64

sudo docker build -f Dockerfile.arm64 -t wxhere/ubuntu-desktop-lxde-vnc:lxqt-arm64 .

docker run -d --rm \
  --name=ubuntu-desktop-lxqt-test \
  -e VNC_PASSWORD=123456 \
  -e HTTP_PASSWORD=123456 \
  -e USER=sandbox \
  -e PASSWORD=123456 \
  -p 5900:5900 \
  -p 6080:80 \
  wxhere/ubuntu-desktop-lxde-vnc:lxqt-arm64

docker manifest create wxhere/ubuntu-desktop-lxde-vnc:lxqt ubuntu-desktop-lxde-vnc:lxqt-amd64 ubuntu-desktop-lxde-vnc:lxqt-arm64
docker manifest annotate wxhere/ubuntu-desktop-lxde-vnc:lxqt ubuntu-desktop-lxde-vnc:lxqt-amd64 --os linux --arch amd64
docker manifest annotate wxhere/ubuntu-desktop-lxde-vnc:lxqt ubuntu-desktop-lxde-vnc:lxqt-arm64 --os linux --arch arm64
docker manifest push ubuntu-desktop-lxde-vnc:lxqt
